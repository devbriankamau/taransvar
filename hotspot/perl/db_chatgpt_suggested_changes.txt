#!/bin/bash
set -e

IPT=/sbin/iptables
WAN=enp1s0
LAN=wlp2s0
LAN_NET=192.168.50.0/24

# --- reset ---
$IPT -F
$IPT -t nat -F
$IPT -X

# default policies
$IPT -P INPUT DROP
$IPT -P OUTPUT DROP
$IPT -P FORWARD DROP

# loopback
$IPT -A INPUT  -i lo -j ACCEPT
$IPT -A OUTPUT -o lo -j ACCEPT

# established
$IPT -A INPUT   -m conntrack --ctstate ESTABLISHED,RELATED -j ACCEPT
$IPT -A OUTPUT  -m conntrack --ctstate ESTABLISHED,RELATED -j ACCEPT
$IPT -A FORWARD -m conntrack --ctstate ESTABLISHED,RELATED -j ACCEPT

# --- LAN services on router (DHCP + portal HTTP on router) ---
# DHCP server replies to clients
$IPT -A INPUT  -i $LAN -p udp --sport 68 --dport 67 -j ACCEPT
$IPT -A OUTPUT -o $LAN -p udp --sport 67 --dport 68 -j ACCEPT

# Captive portal web server on router (port 80)
$IPT -A INPUT  -i $LAN -p tcp --dport 80 -j ACCEPT
$IPT -A OUTPUT -o $LAN -p tcp --sport 80 -j ACCEPT

# (optional but recommended) allow router itself to do DNS + web out (updates etc.)
$IPT -A OUTPUT -o $WAN -p udp --dport 53 -j ACCEPT
$IPT -A OUTPUT -o $WAN -p tcp --dport 53 -j ACCEPT
$IPT -A OUTPUT -o $WAN -p tcp --dport 80  -j ACCEPT
$IPT -A OUTPUT -o $WAN -p tcp --dport 443 -j ACCEPT

# --- allowed client set ---
ipset create allowed_lan hash:ip -exist
ipset flush allowed_lan
ipset add allowed_lan 192.168.50.104 -exist

# --- captive portal hijack: ONLY for NOT-allowed clients ---
# If allowed: do not hijack
$IPT -t nat -I PREROUTING 1 -i $LAN -p tcp --dport 80 \
  -m set --match-set allowed_lan src -j RETURN

# If not allowed: hijack HTTP to router:80
$IPT -t nat -A PREROUTING -i $LAN -p tcp --dport 80 -j REDIRECT --to-ports 80

# --- forwarding: ONLY allowed clients can go out ---
$IPT -A FORWARD -i $LAN -o $WAN -s $LAN_NET \
  -m set --match-set allowed_lan src \
  -m conntrack --ctstate NEW,ESTABLISHED,RELATED -j ACCEPT

# return traffic is already covered by FORWARD ESTABLISHED rule above

# --- NAT: only for LAN subnet (or only allowed set if you prefer) ---
$IPT -t nat -A POSTROUTING -s $LAN_NET -o $WAN -j MASQUERADE

# If you don't want that, omit these two lines.
